الكود 4

Single Neuron Learning (Mean Error)

X = [1, 2, 3]
Y = [2, 4, 6]

w = 0.0
lr = 0.1

for _ in range(10):
    preds = [w*x for x in X]
    errors = [y - p for y, p in zip(Y, preds)]
    mean_error = sum(errors) / len(errors)
    w += lr * mean_error

print("Learned weight:", w)

الكود 8

BAM with Two Pattern Pairs

import numpy as np

X = np.array([[1, -1],
              [-1, 1]])

Y = np.array([[1, 1],
              [-1, -1]])

W = np.zeros((2, 2))

for i in range(2):
    W += np.outer(X[i], Y[i])

test_X = np.array([1, -1])
Y_out = np.sign(test_X @ W)

print(W)
print(Y_out)
