البحث الخطي (Linear Search)

def linear_search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1

# استخدام
arr = [4, 2, 7, 1]
print(linear_search(arr, 7))  # الناتج: 2

شجرة البحث الثنائية (BST – Insertion + Search)

class Node:
    def __init__(self, key):
        self.data = key
        self.left = None
        self.right = None

def insert(root, key):
    if root is None:
        return Node(key)
    if key < root.data:
        root.left = insert(root.left, key)
    else:
        root.right = insert(root.right, key)
    return root

def search(root, key):
    if root is None or root.data == key:
        return root
    if key < root.data:
        return search(root.left, key)
    return search(root.right, key)

# استخدام
root = None
root = insert(root, 50)
insert(root, 30)
insert(root, 70)
print(search(root, 70) is not None)  # True

القائمة المرتبطة (Linked List – Insert + Display)

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def insert_at_end(self, data):
        node = Node(data)
        if self.head is None:
            self.head = node
        else:
            temp = self.head
            while temp.next:
                temp = temp.next
            temp.next = node

    def display(self):
        temp = self.head
        while temp:
            print(temp.data, end=" -> ")
            temp = temp.next
        print("None")

# استخدام
ll = LinkedList()
ll.insert_at_end(10)
ll.insert_at_end(20)
ll.display()  # 10 -> 20 -> None

التكرار (Recursion)

أ. المضروب (Factorial)

class FactorialCalculator:
    def factorial(self, n):
        if n <= 1:
            return 1
        return n * self.factorial(n - 1)

# استخدام
f = FactorialCalculator()
print(f.factorial(5))  # 120

ب. جمع الأعداد الطبيعية

class RecursiveSum:
    def sum(self, n):
        if n <= 0:
            return 0
        return n + self.sum(n - 1)

# استخدام
s = RecursiveSum()
print(s.sum(5))  # 15

ج. القوة (Power)

class PowerCalculator:
    def power(self, base, exp):
        if exp == 0:
            return 1
        return base * self.power(base, exp - 1)

# استخدام
p = PowerCalculator()
print(p.power(2, 3))  # 8

خوارزمية الترتيب السريع (QuickSort)

class QuickSort:
    def quicksort(self, arr):
        if len(arr) <= 1:
            return arr
        pivot = arr[len(arr) // 2]
        left = [x for x in arr if x < pivot]
        middle = [x for x in arr if x == pivot]
        right = [x for x in arr if x > pivot]
        return self.quicksort(left) + middle + self.quicksort(right)

# استخدام
arr = [5, 2, 8, 10, 3, 0, 4]
qs = QuickSort()
print(qs.quicksort(arr))  # [0, 2, 3, 4, 5, 8, 10]

تشفير قيصر (String-Based Encryption - Python)

def encrypt(text, key):
    result = ""
    for char in text:
        if char.isupper():
            result += chr((ord(char) - 65 + key) % 26 + 65)
        elif char.islower():
            result += chr((ord(char) - 97 + key) % 26 + 97)
        else:
            result += char
    return result

# استخدام
print(encrypt("HelloWorld", 3))  # KhoorZruog

فك تشفير قيصر (Python)

def decrypt(text, key):
    result = ""
    for char in text:
        if char.isupper():
            result += chr((ord(char) - 65 - key + 26) % 26 + 65)
        elif char.islower():
            result += chr((ord(char) - 97 - key + 26) % 26 + 97)
        else:
            result += char
    return result

# استخدام
print(decrypt("KhoorZruog", 3))  # HelloWorld

تشفير ASCII مباشر (Python)

def ascii_encrypt(text, key):
    encrypted = ""
    for char in text:
        if char.isalpha():
            shifted = ord(char) + key
            if char.isupper() and shifted > ord('Z'):
                shifted -= 26
            elif char.islower() and shifted > ord('z'):
                shifted -= 26
            encrypted += chr(shifted)
        else:
            encrypted += char
    return encrypted

# استخدام
print(ascii_encrypt("CyberSecurity", 4))  # GcfiwWigvymx
