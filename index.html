2. إنشاء SCHEMA و View و Index

CREATE SCHEMA schema_name;

CREATE TABLE schema_name.table_name (
 column1_name column1_data_type,
 column2_name column2_data_type,
 column3_name column3_data_type,
 PRIMARY KEY (primary_key_column)
);

CREATE VIEW sales.customer_orders AS
SELECT customers.first_name, customers.last_name,
orders.order_date, orders.total_amount
FROM sales.customers
JOIN sales.orders ON customers.customer_id = orders.customer_id;

CREATE INDEX idx_customer_name ON sales.customers (last_name, first_name);

3. مختبر SQL Injection

CREATE DATABASE SqlInjectionLab;
USE SqlInjectionLab;

CREATE TABLE Users (
 UserID INT IDENTITY(1,1) PRIMARY KEY,
 Username NVARCHAR(50),
 Password NVARCHAR(50)
);

INSERT INTO Users (Username, Password) VALUES ('admin', 'admin123');

-- Example of vulnerable query
DECLARE @Username NVARCHAR(50) = 'admin';
DECLARE @Password NVARCHAR(50) = 'admin123';
SELECT * FROM Users WHERE Username = '' OR 1=1 --' AND Password = 'anyPassword';

-- Union Injection
DECLARE @Username NVARCHAR(50) = ''' UNION SELECT null, null, null --';
DECLARE @Password NVARCHAR(50) = 'anyPassword';

-- Boolean-Based Injection
DECLARE @Username NVARCHAR(50) = ''' AND 1=1 --';
DECLARE @Password NVARCHAR(50) = 'anyPassword';

-- Safe usage with parameters
DECLARE @Username NVARCHAR(50) = 'admin';
DECLARE @Password NVARCHAR(50) = 'admin123';
SELECT * FROM Users WHERE Username = @Username AND Password = @Password;
